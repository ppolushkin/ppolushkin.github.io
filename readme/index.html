public static void mergeSort(int[] a) {
    mergeSort(a, 0, a.length - 1);
}

static void mergeSort(int[] a, int low, int high) {
    if (low == high) {
        return;
    }

    int middle = (low + high) >>> 1;

    mergeSort(a, low, middle);
    mergeSort(a, middle + 1, high);
    merge(a, low, middle, high);
}
